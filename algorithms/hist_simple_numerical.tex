\begin{algorithm}[H]\fontfamily{lmss}
\caption{Naive Privacy Preserving 1D Histogram for Numerical Values}\label{a:1d-simple-histogram-numerical}
\begin{algorithmic}[1]
\renewcommand{\algorithmicrequire}{\textbf{Private Vars:}}
\Require $\red{array}, \red{min}, \red{max}, \red{histogram}, \red{cellWidth}, \red{cell}$

\Procedure{1DimSimpleHistogramNumerical}{$\red{array}[N], cells, \red{min}, \red{max}$}

    \State $\red{cellWidth} \gets (\red{max} \hminus \red{min}) \hdiv cells $\Comment{{\small $\red{cellWidth}$ gets a float value}}

    \For{$i \in \{0, \dots, N-1\}$}

        \State $\red{cell} \gets \red{array}[i] \hdiv \red{cellWidth}$\Comment{{\small $\red{cell}$ gets an encrypted value in range $[0, cells-1]$}}

        \For{$j \in \{0, \dots, cells-1\}$}

            \State $\red{histogram}[j] \gets \red{histogram}[j] \hplus (\red{cell} \heq j)$\Comment{{\small Add ${\enc{1}}$ if $\red{cell}$ is equal to $j$, \par\hfill ${\enc{0}}$ otherwise}}

        \EndFor
    \EndFor
    \State \textbf{return} {$\red{histogram}$}
\EndProcedure
\end{algorithmic}
\end{algorithm}
