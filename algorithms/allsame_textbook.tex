\begin{algorithm}[H]\fontfamily{lmss}
\caption{All Examples Same Textbook Algorithm}\label{a:id3-same-simple}
\begin{algorithmic}[1]
\renewcommand{\algorithmicrequire}{\textbf{Global Vars:}}
\Require $classAttribute$
\Procedure{AllExamplesSame}{$examples, attribute$}\Comment{{\small Check if all examples are \par\hfill of the same class
}}
    \For{$i \in \{0, \dots, \textsc{Length}(examples)-2\}$}\Comment{{\small Check all examples, two at a time}}
        \State $ex_1 \gets examples[i]$
        \State $ex_2 \gets examples[i+1]$

        \If{$(ex_1[classAttribute] \neq ex_2[classAttribute])$}
            \State \textbf{return} $false$
        \EndIf
    \EndFor
    \State \textbf{return} $true$
\EndProcedure
\end{algorithmic}
\end{algorithm}
