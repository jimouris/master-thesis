\begin{algorithm}[H]\fontfamily{lmss}
\caption{C4.5 Textbook Algorithm}\label{a:c45-simple}
\begin{algorithmic}[1]
\renewcommand{\algorithmicrequire}{\textbf{Global Vars:}}
\Require $classAttribute$
\Procedure{C45}{$examples, attributes$}
    \If   {$attributes = \{\}$}\Comment{{\small If number of predicting attributes is empty}}
          \State \textbf{return} $\textsc{MostCommonLabel}(examples)$\Comment{{\small Return the label with the most common \par\hfill value of the target attribute in the examples}}
    \ElsIf {$\textsc{AllExamplesSame}(examples)$}
        \State \textbf{return} $examples[0][classAttribute]$\Comment{{\small Return a node with the label that is common \par\hfill to all examples}}
    \EndIf


    \State $(bestAttribute, bestThreshold, bestSplitted) \gets \textsc{Best}(examples, attributes)$\Comment{{\small Best \par\hfill attribute is the one whose split provides maximum information gain. Best threshold is  \par\hfill where that split happens. $bestSplitted$ corresponds to the subsets generated by that split. }}

    \State $branches \gets \{\}$
    \If {$bestAttribute \in categoricalAttributes$}
      \For{{\small each possible value} $v_i$ {\small of} $bestAttribute$}
          \State {\small Let} $branch$ {\small be a new tree branch below root, corresponding to the test}  \par\hfill $bestAttribute = v_i$
          \State $subset \gets bestSplitted[v_i]$

          \If {$subset = \{\}$}
              \State $branch \gets \textsc{AddLeaf}(branch, \textsc{MostCommonLabel}(examples)))$
          \Else
              \State $branch \gets \textsc{AddTree}(branch, \textsc{C45}(subset,$ $attributes - \{bestAttribute\}))$\par\hfill
          \EndIf
          \State $branches \gets branches \cup branch$
      \EndFor
    \Else
      \State {\small Let} $branch$ {\small be a new tree branch below root, corresponding to the test}  \par\hfill $bestAttribute <= v_i$
      \State $less \gets bestSplitted[0]$
      \If {$less = \{\}$}
          \State $branch \gets \textsc{AddLeaf}(branch, \textsc{MostCommonLabel}(examples)))$
      \Else
          \State $branch \gets \textsc{AddTree}(branch, \textsc{C45}(less,$ $attributes - \{bestAttribute\}))$\par\hfill
      \EndIf
      \State $branches \gets branches \cup branch$
      \State {\small Let} $branch$ {\small be a new tree branch below root, corresponding to the test}  \par\hfill $bestAttribute > v_i$
      \State $greater \gets bestSplitted[1]$
      \If {$greater = \{\}$}
          \State $branch \gets \textsc{AddLeaf}(branch, \textsc{MostCommonLabel}(examples)))$
      \Else
          \State $branch \gets \textsc{AddTree}(branch, \textsc{C45}(greater,$ $attributes - \{bestAttribute\}))$\par\hfill
      \EndIf
      \State $branches \gets branches \cup branch$
    \EndIf
    \State \textbf{return} $branches$
\EndProcedure
\end{algorithmic}
\end{algorithm}
