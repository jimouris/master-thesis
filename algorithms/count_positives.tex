\begin{algorithm}[H]\fontfamily{lmss}
\caption{Privacy Preserving Count Positives Algorithm}\label{a:id3-count-positives-pp}
\begin{algorithmic}[1]
\Require $\red{examples}, \red{neq}, \red{count}$
\Procedure{CountPositives}{$\red{examples}$}

  \State $\red{neq} \gets \red{examples} \hneq \enc{-1}$\Comment{{\small {\texttt{SIMD} comparison}}}
  \State $\red{count} \gets \textsc{Sum}(\red{neq})$
  \State \textbf{return} $\red{count}$
\EndProcedure
\end{algorithmic}
\end{algorithm}
